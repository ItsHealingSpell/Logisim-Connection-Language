LCL Format
IN number_of_inputs; (identifier:input[n])
OUT number_of_outputs; (identifier:output[n])
NOT number_of_nots; (identifier:not[n])
AND number_of_ands; (identifier:and[n])
XOR number_of_xors; (identifier:xor[n])
OR number_of_ors; (identifier:or[n])
CNCT (identifier of the component that's providing the bit)[which_output],(identifier of the component that's using the bit)[which_input];
